sum[i][j] = sum [i][j-1] + sum[i-1][j] - sum[i-1][j-1] + a[i][j]

#include<stdio.h>
const int N=1e3+5;
int sum[N][N];
int a[N][N];
int n,m;
void cal(){
    for(int i = 1;i <= n;i++){
        for(int j = 1;j <= m;j++)
            sum[i][j] = sum [i][j-1] + sum[i-1][j] - sum[i-1][j-1] + a[i][j];
    }
}
int main(){
    int n,m,q;
    scanf("%d%d%d",&n,&m,&q);
    for(int i = 1;i <= n;i++){
        for(int j = 1;j <= m;j++)
            scanf("%d",&a[i][j]);
    }
    while (q --){
         int x1,y1,x2,y2;
         scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
         cal();
         int res = sum[x2][y2] - sum[x1][y1];
         printf("%d\n",res);
    }    
    return 0;
}
